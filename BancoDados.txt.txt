create database BancoProjeto;

CREATE TABLE bancoWeb (
  `idCliente` INT(11) NOT NULL AUTO_INCREMENT,
  `nome` VARCHAR(45) NULL DEFAULT NULL,
  `endereco` VARCHAR(100) NULL DEFAULT NULL,
  `telefone` BIGINT(20) NULL DEFAULT NULL,
  `pessoaF` TINYINT(4) NULL DEFAULT NULL,
  `pessoaJ` TINYINT(4) NULL DEFAULT NULL,
  `cpf` VARCHAR(45) NULL DEFAULT NULL,
  `cnpj` VARCHAR(45) NULL DEFAULT NULL,
  `nomeMae` VARCHAR(45) NULL DEFAULT NULL,
  `nomePai` VARCHAR(45) NULL DEFAULT NULL,
  `nomeFantasia` VARCHAR(45) NULL DEFAULT NULL,
  `contaCC` TINYINT(4) NULL DEFAULT NULL,
  `contaP` TINYINT(4) NULL DEFAULT NULL,
  `numeroConta` VARCHAR(45) NULL DEFAULT NULL,
  `saldo` DOUBLE NULL DEFAULT NULL,
  `situacao` TINYINT(4) NOT NULL,
  `seguro` TINYINT(4) NULL DEFAULT NULL,
  `numeroSeg` VARCHAR(45) NULL DEFAULT NULL,
  `valorSeg` DOUBLE NULL DEFAULT NULL,
  `situacaoSeg` TINYINT(4) NOT NULL,
  PRIMARY KEY (`idCliente`));
